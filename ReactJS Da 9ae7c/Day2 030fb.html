import React from 'react'
export default function DigitCount() {
    var n = 85;
    var m;
    if(n>70){
        m="Distinct"
    }
    else if(n>60){
        m="First"
    }
    else if (n>50) {
        m="Second"
    } else {
        m="Fail"
    }
    //var c=n.toString.length();
    var num=n;
    var count=0;
    if(num>=1) ++count;
    while(num/10>=1){
        num=num/10
        ++count
    }
    
  return (
    <div>
        <h1>Result of exam </h1>
        <h1>{m}</h1>
        <h1>No of Digits </h1>
        <h1>{count}</h1>
    </div>
  )
}
